2b) lsb image steganogrpahy (code)
from PIL import Image

img = Image.open("your image page").convert("RGB")
pixels = img.load()

message = input("enter the message to be hidden")
msg = message + "#"  

bits = ''.join(format(ord(c), '08b') for c in msg)

i = 0
for y in range(img.height):
    for x in range(img.width):
        if i < len(bits):
            r, g, b = pixels[x, y]
            pixels[x, y] = ((r & ~1) | int(bits[i]), g, b)
            i += 1

img.save("stego.png")
print("Message hidden!")


img = Image.open("stego.png")
pixels = img.load()

bits = ""
for y in range(img.height):
    for x in range(img.width):
        bits += str(pixels[x, y][0] & 1)

msg = ""
for i in range(0, len(bits), 8):
    msg += chr(int(bits[i:i+8], 2))
    if msg.endswith("#"):
        break

print("Hidden message:", msg[:-1])

2c) rsa algorithm
- openssl genrsa -out key.pri 2048
- openssl rsa -in key.pri -out key.pub -pubout
- openssl rsa -in key.pri -noout -text
- cat key.pub
- echo "hello" > message.txt
- openssl rsautl -encrypt -inkey key.pub -pubin -in message.txt -out message.enc
- openssl rsautl -decrypt -inkey key.pri -in message.enc -out message.dec
- openssl rand -hex 32 > secret.key
- openssl rsautl -encrypt -inkey key.pub -pubin -in secret.key -out secret.key.enc
- openssl rsautl -decrypt -inkey key.pri -in secret.key.enc -out secret.key.dec
